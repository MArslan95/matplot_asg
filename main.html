(5) Use plt.subplots to insert two of the above graphs into one figure

In[6]:
fig, axes = plt.subplots(2, 1, figsize=(8, 6))

# Plot a random scatter plot on the first axis
axes[0].scatter(rnd.randn(100), rnd.randn(100), c='red')
axes[0].set_xlabel('X')
axes[0].set_ylabel('Y')
axes[0].set_title('Random Scatter Plot')

# Plot a random bar plot on the second axis
axes[1].bar(rnd.randint(10, size=10), rnd.randint(10, size=10))
axes[1].set_xlabel('X')
axes[1].set_ylabel('Y')
axes[1].set_title('Random Bar Plot')

fig.suptitle('Two Random Graphs in One Figure')
plt.show()

(7) Determine an arbitrary function and create a function that differentiates it. Display these two together on one graph and write down what the differential value means.

In[8]:
# Define an arbitrary function
def f(x):
  return x**2 + 2*x + 1

# Define a function that differentiates the given function
def f_prime(x):
  return 2*x + 2

# Generate a range of x-values
x = np.linspace(-5, 5, 100)

# Calculate the function values and its derivative
y = f(x)
y_prime = f_prime(x)

# Plot the function and its derivative on the same graph
plt.figure(figsize=(8, 6))
plt.plot(x, y, label='f(x)')
plt.plot(x, y_prime, label='f\'(x)')
plt.xlabel('X')
plt.ylabel('Y')
plt.title('Function and Derivative')
plt.legend()
plt.show()
